$cmdhome = "d:/dev/bin/cmd" # is it possible to calculate this?

# https://github.com/kelleyma49/PSFzf/blob/master/PSFzf.Functions.ps1
function j([string] $alias) {
    $aliasfile = "${cmdhome}/util/shell/.wcd/alias.wcd"
    cat $aliasfile | fzf -i +s -n1 -e -q $alias -0 -1 | foreach-object { $result = $_ }
    if ($result -ne $null) {
        $dst = ($result -split " ")[1]
        set-location $dst
    }
}

get-childitem -path $cmdhome -recurse | where psiscontainer | where {$_.FullName -notlike "*\.*"} |
foreach-object {
    $dirname = $_.FullName; $env:PATH="${env:PATH};${dirname}"
}
