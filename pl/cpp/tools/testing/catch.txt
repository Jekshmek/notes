int
add(int a, int b) {
    return a + b;
}

#define CATCH_CONFIG_MAIN
#include <catch.hpp>

TEST_CASE("should add two numbers", "[add]") { // [...] - is a tag
    REQUIRE(add(1, 2) == 3);
}
